class UMLDocument {
  - shapes : List<Shape>
  - lines : List<Line>
  - listeners : List<Runnable>
  + addListener(listener: Runnable)
  + notifyListeners()
  + getShapes() : List<Shape>
  + getLines() : List<Line>
  + addShape(s: Shape)
  + addLine(l: Line)
  + moveShapeToFront(s: Shape)
  + groupSelectedShapes()
  + ungroupSelectedShapes()
}
note left of UMLDocument
  **notifyListeners()**:
    foreach listener in listeners: listener.run()

  **addShape(s) / addLine(l)**:
    shapes.add(s) / lines.add(l)
    notifyListeners()

  **groupSelectedShapes()**:
    selected = filter(shapes, s -> s.isSelected)
    if selected.size >= 2:
      group = new CompositeShape(selected)
      shapes.removeAll(selected)
      shapes.add(group)
      notifyListeners()
      
  **ungroupSelectedShapes()**:
    selected = filter(shapes, s -> s.isSelected)
    if selected.size == 1 and selected[0] is CompositeShape:
      shapes.remove(selected[0])
      shapes.addAll(selected[0].getChildren())
      notifyListeners()
end note

note left of Canvas
  **Canvas(document)**:
    this.document = document
    document.addListener(() -> repaint())

  **setMode(mode)**:
    this.currentMode = mode

  **getShapes()**:
    return document.getShapes()

  **addShape(s) / addLine(l)**:
    document.addShape(s) / document.addLine(l)
    // repaint() 由 Observer Pattern 自動處理

  **setPreviewShape(s)**:
    this.previewShape = s
    repaint() // 預覽屬於 View 狀態，直接 repaint

  **paintComponent(g)**:
    foreach line in document.getLines(): line.draw(g)
    foreach shape in document.getShapes(): shape.draw(g)
    if previewShape != null: previewShape.draw(g)
end note

abstract class Shape {
  # x, y, w, h : int
  # isSelected : boolean
  # isHovered : boolean
  # ports : Port[]
  + draw(g: Graphics)
  + contains(p: Point) : boolean
  + move(dx: int, dy: int)
  + setSelected(b: boolean)
  + isSelected() : boolean
  + setHovered(b: boolean)
  + isHovered() : boolean
  + setBounds(x: int, y: int, w: int, h: int)
  + getBounds() : Rectangle
  + getPortAt(p: Point) : Port
  + getPortLocation(index: int) : Point
  + resizeByPort(...)
}
note left of Shape
  **draw(g)**:
    abstract method (impl by subclasses)
  
  **contains(p)**:
    return bounds(x, y, w, h).contains(p)

  **move(dx, dy)**:
    this.x += dx; this.y += dy

  **setSelected(b)**:
    this.isSelected = b

  **getPortAt(p)**:
    foreach port in ports:
      if port.contains(p) return port
    return null

  **getPortLocation(index)**:
    return Point(calc_X, calc_Y) based on index

  **resizeByPort(portIndex, origBounds, newPos)**:
    calc newX, newY, newW, newH
    this.setBounds(newX, newY, newW, newH)
end note

class SelectMode {
  + mousePressed(e: MouseEvent)
  + mouseDragged(e: MouseEvent)
  + mouseReleased(e: MouseEvent)
  + mouseMoved(e: MouseEvent)
}
note right of SelectMode
  **mouseMoved(e)**:
    foreach shape in canvas.getShapes():
      shape.setHovered(shape.contains(e.getPoint()))
    repaint()

  **mousePressed(e)**:
    if 點到 Port: state = RESIZING
    else if 點到 Shape: state = MOVING
    else: state = SELECT_BOX

  **mouseDragged(e)**:
    if state == RESIZING:
      resizingShape.resizeByPort(...)
    else if state == MOVING:
      foreach shape in selectedShapes: shape.move(dx, dy)
    else if state == SELECT_BOX:
      canvas.setSelectionBox(newRect)

  **mouseReleased(e)**:
    if state == SELECT_BOX:
      foreach shape in canvas.getShapes():
        if box.contains(shape) shape.setSelected(true)
      canvas.setSelectionBox(null)
    state = IDLE
end note

class CreateShapeMode {
  + processExternalDrag(e: MouseEvent)
  + processExternalRelease(e: MouseEvent)
}
note bottom of CreateShapeMode
  **processExternalDrag(e)**:
    preview.setBounds(e.getPoint())
    canvas.setPreviewShape(preview)

  **processExternalRelease(e)**:
    if canvas.contains(e.getPoint()):
      canvas.addShape(preview)
    canvas.setPreviewShape(null)
    canvas.setMode(previousMode)
end note

class CreateLinkMode {
  + mousePressed(e: MouseEvent)
  + mouseDragged(e: MouseEvent)
  + mouseReleased(e: MouseEvent)
}
note bottom of CreateLinkMode
  **mousePressed(e)**:
    startPort = canvas.getShapes().getPortAt(e.getPoint())
    if startPort != null:
      previewLine = lineFactory(startPort, null)

  **mouseDragged(e)**:
    if previewLine != null:
      previewLine.setTempEndPoint(e.getPoint())

  **mouseReleased(e)**:
    endPort = canvas.getShapes().getPortAt(e.getPoint())
    if startPort != null and endPort != null and startPort != endPort:
      line = lineFactory(startPort, endPort)
      canvas.addLine(line)
    startPort = null; previewLine = null
end note
